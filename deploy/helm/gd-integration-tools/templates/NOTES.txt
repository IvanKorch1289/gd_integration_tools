{{- " — Internal integration platform" }}

Deploy:
  helm install gd-integration ./deploy/helm/gd-integration-tools \
    --namespace {{ .Values.namespace.name }} \
    --create-namespace \
    --set image.tag={{ .Values.image.tag }}

Upgrade:
  helm upgrade gd-integration ./deploy/helm/gd-integration-tools \
    --namespace {{ .Values.namespace.name }}

Uninstall:
  helm uninstall gd-integration --namespace {{ .Values.namespace.name }}

Verify:
  kubectl get pods -n {{ .Values.namespace.name }}
  kubectl get hpa -n {{ .Values.namespace.name }}
